## `src/utils.js`

```js
// 랜덤 빈 셀 찾기 + 새 블록 추가
export function getEmptyCells(grid){
    const empty=[];
    for(let r=0;r<4;r++){
        for(let c=0;c<4;c++){
            if(grid[r][c]===null) empty.push([r,c]);
        }
    }
    return empty;
}

export function addRandomTile(grid){
    const empty=getEmptyCells(grid);
    if(empty.length===0) return false;
    const [r,c]=empty[Math.floor(Math.random()*empty.length)];
    grid[r][c]=Math.random()<0.9?2:4;
    return true;
}
```

## `src/board.js`

```js
import {addRandomTile} from './utils.js';

export class Board{
    constructor(){
        this.grid=Array.from({length:4},()=>Array(4).fill(null));
    }

    init(){
        // 두 개의 랜덤 블록 생성
        addRandomTile(this.grid);
        addRandomTile(this.grid);
    }

    render(container){
        container.innerHTML='';
        for(let r=0;r<4;r++){
            for(let c=0;c<4;c++){
                const cell=document.createElement('div');
                cell.className='cell';
                if(this.grid[r][c]!==null){
                    cell.textContent=this.grid[r][c];
                    // 색상(숫자별) 옵션을 넣어도 됩니다.
                }
                container.appendChild(cell);
            }
        }
    }

    // 나중에 이동/병합 로직 추가
}
```

## `src/app.js`

```js
import {Board} from './board.js';

const gridEl=document.getElementById('grid');
const restartBtn=document.getElementById('restart-btn');

let board=new Board();

function startGame(){
    board=new Board();
    board.init();
    board.render(gridEl);
}
startGame();

// 재시작 버튼
restartBtn.addEventListener('click',()=>{
    startGame();
});
```
